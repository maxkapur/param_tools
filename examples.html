np.random.seed(547)

def spiral(t):
    return np.array([np.exp(-.1*t)*np.cos(t), np.exp(-.1*t)*np.sin(t)])

domain = [0, 5*np.pi]

# Random points
x, t, s = r_arc(200, spiral, *domain, 50)

# Shape of spiral
shape_t = np.linspace(*domain, 500)
shape_x = spiral(shape_t)

fig, ax = plt.subplots(figsize=(8,8))
ax.plot(*shape_x, color='gray', ls="--")
ax.scatter(*x, marker='+', alpha=0.7, color='forestgreen')

ax.set_xlim(-.85, 1.1)
ax.set_ylim(-.85, 1.1)

spiral_A = arc_length(spiral, 0, 5*np.pi, 500)

ax.set_title('Total arc length: {}'.format(np.round(spiral_A, 3)), y=0.02, x=.97, loc='right')

plt.savefig('spiral.png')

# Function and parameter region
def wave(t, u):
    return np.array([t, u, np.cos(t)*np.cos(u/1.4)])

domain_t = [0, 2*np.pi]
domain_u = [0, 2*np.pi]

# Get random points
x, t, u, St, Su = r_surface(3000, wave, *domain_t, *domain_u, 50,50)

# Set up plot
fig = plt.figure(figsize=(10,10))
fig.subplots_adjust(hspace=0, wspace=0.1, bottom=.2)
ax = fig.add_subplot(111, projection='3d')
ax.view_init(30, 20)
ax.set_xlim(domain_t)
ax.set_ylim(domain_u)
ax.set_zlim(-np.pi, np.pi)

# Plot random points
ax.scatter(*x, marker='+', alpha=0.5, color='cornflowerblue')

# Plot the function itself
shape_t, shape_u = np.meshgrid(np.linspace(*domain_t, 25), np.linspace(*domain_u, 25))
shape_x = wave(shape_t, shape_u)
ax.plot_wireframe(*shape_x, color='black', ls="--", alpha = 0.3)

# Surface area
wave_A = surface_area(wave, *domain_t, *domain_u, 100, 100)
ax.set_title('Total surface area: {}'.format(np.round(wave_A, 3)), size=15, y=-0.05)
